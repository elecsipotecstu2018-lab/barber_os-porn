    <script>
        // กำหนด URL ของ Google Apps Script ที่คุณจะนำมาใส่ในภายหลัง
        const SCRIPT_URL = https://script.google.com/macros/s/AKfycbwCCjgMawBKtPEOAiLGd0_1YObymZgV89TbXg1Wkc2r5coI8Nr5ZJAaGJe_8SwUSpGyYw/exec

        // ตั้งค่าให้ช่อง 'วันที่และเวลา' แสดงเวลาปัจจุบันอัตโนมัติเมื่อเปิดหน้าเว็บ
        window.addEventListener('load', () => {
            const now = new Date();
            now.setMinutes(now.getMinutes() - now.getTimezoneOffset());
            document.getElementById('dateTime').value = now.toISOString().slice(0,16);
        });

        // ดักจับเหตุการณ์เมื่อมีการกดปุ่ม Submit (ส่งฟอร์ม)
        const form = document.getElementById('vendingForm');
        const submitBtn = document.getElementById('submitBtn');

        form.addEventListener('submit', e => {
            e.preventDefault(); // ป้องกันไม่ให้หน้าเว็บรีเฟรช

            // แสดงหน้าต่างโหลดให้ผู้ใช้ทราบว่าระบบกำลังทำงาน
            submitBtn.classList.add('loading');
            submitBtn.innerHTML = 'กำลังบันทึกข้อมูล... ⏳';

            // ดึงข้อมูลทั้งหมดจากฟอร์ม
            let formData = new FormData(form);

            // ส่งข้อมูลไปยัง Google Apps Script
            fetch(SCRIPT_URL, { method: 'POST', body: formData })
                .then(response => {
                    // เมื่อบันทึกสำเร็จ ให้แสดง Pop-up แจ้งเตือน
                    Swal.fire({
                        icon: 'success',
                        title: 'บันทึกสำเร็จ!',
                        text: 'ข้อมูลของคุณถูกบันทึกลงในระบบเรียบร้อยแล้ว',
                        confirmButtonColor: '#4CAF50'
                    });
                    form.reset(); // ล้างข้อมูลในฟอร์ม
                    
                    // เซ็ตเวลาปัจจุบันใหม่อีกครั้งหลังล้างฟอร์ม
                    const now = new Date();
                    now.setMinutes(now.getMinutes() - now.getTimezoneOffset());
                    document.getElementById('dateTime').value = now.toISOString().slice(0,16);
                })
                .catch(error => {
                    // กรณีเกิดข้อผิดพลาด
                    Swal.fire({
                        icon: 'error',
                        title: 'เกิดข้อผิดพลาด',
                        text: 'ไม่สามารถบันทึกข้อมูลได้ กรุณาลองใหม่อีกครั้ง หรือตรวจสอบ SCRIPT_URL',
                        confirmButtonColor: '#d33'
                    });
                    console.error('Error!', error.message);
                })
                .finally(() => {
                    // คืนค่าปุ่มให้กลับมาเป็นเหมือนเดิม
                    submitBtn.classList.remove('loading');
                    submitBtn.innerHTML = '💾 บันทึกข้อมูล';
                });
        });
    </script>
